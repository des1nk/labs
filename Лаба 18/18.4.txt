#include <iostream>
#include <string>
#include <math.h>
using namespace std;
int main()
{
	int m, n, i, j, x;
	float sum;
	cout << "M = ";
	cin >> m;
	cout << "N = ";
	cin >> n;
	cout << "--------\n";
	int** Mas = new int* [m];
	for (i = 0; i < m; i++)
		Mas[i] = new int[n];

	for (i = 0; i < m; i++)
		for (j = 0; j < n; j++)
		{
			Mas[i][j] = rand() % 10;
		}
	for (i = 0; i < m; i++)
	{
		for (j = 0; j < n; j++)
			cout << Mas[i][j] << " ";
		cout << "\n";
	}
	cout << "--------\n";
	
	for (j = 0; j < n; j++)
	{
		x = 0;
		sum = 0;
		for (i = 0; i < m; i++)
		{
			sum = sum + Mas[i][j];
		}
		sum = sum / m;
		for (i = 0; i < m; i++)
		{
			if (Mas[i][j] > sum)
			{
				x = x + 1;
			}
		}
		cout << "x(" << j << ") = " << x << endl;
	}
	cout << "--------\n";
	for (i = 0; i < m; i++)
	delete[] Mas[i];
	delete[] Mas;
	return 0;
}
