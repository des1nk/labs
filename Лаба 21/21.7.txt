#include <iostream>
#include <string>
#include <windows.h>
using namespace std;
int main()
{
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	char name[12];
	int i, m, j;
	string S, I;
	cout << "Введите строку-предложение - ";
	getline(cin, S);
	j = 0;
	m = S.size();
	I = S;
	for (i = 0; i < m; i++)
	{
		if (i % 2 != 0) {
			S[j] = I[i];
			j++;
		};
	};
	for (i = (m - 1); i >= 0; i--) {
		if (i % 2 == 0) {
			S[j] = I[i];
			j++;
		};
	};
	cout << "Зашифрованное предложение - " << S;

	return 0;

}
