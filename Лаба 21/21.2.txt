#include <string>
#include <iostream>
#include <windows.h>
using namespace std;
int main()
{
	SetConsoleCP(1251);
	SetConsoleOutputCP(1251);
	char name[12];
	int k, m, i, min, i1;
	string s;
	k = 0;
	cout << "¬ведите строку - ";
	getline(cin, s);
	m = s.size();

	for (i = 0; i < m; i++) {
		if (s[i] != ' ') {
			k++;
		}
		if ((s[i] == ' ') || (i == m - 1)) {
			if (k != 1) {
				min = k;
				i1 = k + 1;
				k = 1;
			}
		}
	}

	for (i = i1; i < m; i++) {
		if (s[i] != ' ') {
			k++;
		}
		if ((s[i] == ' ') || (i == m - 1)) {
			if ((k != 0) & (k < min)) {
				min = k;
			}
			k = 0;
		}
	}

}
