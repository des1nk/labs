#include <string>
#include <iostream>
#include <windows.h>
using namespace std;
int main()
{
	setlocale(LC_ALL, "Russian");
	int l, i;
	char c;
	string S;
	cout << "Введите строку - ";
	getline(cin, S);
	l = S.size();
	cout << "Введите символ - ";
	cin >> c;
	for (i = (l - 1); i >= 0; i--)
		if (S[i] == c)
			S.insert(i, 1, c);
	cout << "Итоговая строка - " << S;

	return 0;

}
